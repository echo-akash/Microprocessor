code segment
assume cs:code, es:code, ds:code,ss:code
ORG 1000H

CALL ALLCLR
CALL CURSR1

MOV SI,OFFSET AB
CALL STRING

X: CALL DELAY
ALLCLR:
MOV AH,00000001B
JMP OUT
CURSR1:
MOV AH,85H


OUT:
CALL BUSY
MOV AL,AH
MOV DX,00H
OUT DX,AL
RET

BUSY:
MOV DX,02H
IN AL,DX
TEST AL,80H
JNZ BUSY
RET

STRING:
MOV AH,CS:[SI]
CMP AH,00H
JE RETURN

CALL BUSY

MOV AL,AH
MOV DX,04H
OUT DX,AL
INC SI
JMP STRING


RETURN:
RET

DELAY:
MOV CX,0FFFFH
F:
DEC CX
JNZ F
JMP X




AB DB 'MIST',00H
AC DB 'CSE JOSS',00H

code ends
end
